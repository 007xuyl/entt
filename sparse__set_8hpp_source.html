277  virtual void destroy(entity_type entity) {
278  assert(has(entity));
279  const auto entt = entity & traits_type::entity_mask;
280  const auto back = direct.back() & traits_type::entity_mask;
281  const auto pos = reverse[entt] & ~in_use;

283  // has size 1), switching the two lines below doesn't work as expected
284  reverse[back] = pos | in_use;
